'use client';

export function Banner({ banner }) {
  const { user } = useAuth();
  const { trackView, trackClick } = useBannerTracking();

  useEffect(() => {
    if (user?.uid) trackView(banner.bannerId, user.uid);
  }, [banner.bannerId, user?.uid, trackView]);

  const handleClick = async () => {
    if (user?.uid) {
      await trackClick(banner.bannerId, user.uid, { clickUrl: banner.linkTo });
    }
    window.location.href = banner.linkTo;
  };

  return <img src={banner.imageUrl} onClick={handleClick} />;
}
